  function formatValue(v) {
    if (v === null || v === undefined) return "";
    const n = Number(v);
    if (Number.isFinite(n) && String(v).trim() !== "") return n.toLocaleString();
    return String(v);
  }

  function renderSection(title, obj) {
    let html = `<div class="card"><h3 style="margin:0 0 10px 0;">${title}</h3>`;
    for (const [k, v] of Object.entries(obj)) {
      html += `<div class="row"><b>${k}</b><span>${formatValue(v)}</span></div>`;
    }
    html += `</div>`;
    return html;
  }

  async function generate() {
    const empId = document.getElementById("empId").value.trim();
    const payDate = document.getElementById("payDate").value; // YYYY-MM-DD
    const msg = document.getElementById("msg");
    const out = document.getElementById("out");

    msg.textContent = "";
    msg.className = "msg";
    out.innerHTML = "";

    if (!empId || !payDate) {
      msg.textContent = "Please enter Employee ID and Pay Date.";
      msg.className = "msg error";
      return;
    }

    const url = new URL(API_URL);
    url.searchParams.set("empId", empId);
    url.searchParams.set("payDate", payDate);

    try {
      msg.textContent = "Generating...";
      const res = await fetch(url.toString());
      const data = await res.json();

      if (!data.ok) {
        msg.textContent = data.error || "No record found.";
        msg.className = "msg error";
        return;
      }

      msg.textContent = "";
      const r = data.result;

      out.innerHTML =
        renderSection("Employee Information", r.Employee_Information) +
        `<div class="grid">` +
          renderSection("Earnings (Per Payday)", r["Earnings_(Per_Payday)"]) +
          renderSection("Deductions (Employee Share)", r["Deductions_(Employee_Share)"]) +
        `</div>` +
        renderSection("NET PAY", r["NET_PAY"]) +
        renderSection("Employer Contributions (Company Expense)", r["Employer_Contributions_(Company_Expense)"]);

    } catch (e) {
      msg.textContent = "Cannot connect to payroll system.";
      msg.className = "msg error";
    }
  }
